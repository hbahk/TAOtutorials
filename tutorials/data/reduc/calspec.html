from astropy.table import Table
import numpy as np
from astropy.coordinates import SkyCoord
from astropy import units as u

calspec = Table.read('calspec.dat', format='ascii.no_header', delimiter=',')

ra = calspec['col2']
dec = calspec['col3']
name = calspec['col1']

c = SkyCoord(ra=ra, dec=dec, unit=(u.hourangle, u.deg))

new_calspec = Table([c.ra.deg, c.dec.deg, name], names=('RA', 'Dec', 'name'))

new_calspec.write('calspec.csv', format='ascii.csv', delimiter=',', overwrite=True)

calspec_mag = Table.read('calspec_mag.dat', format='ascii.csv', delimiter=',')
calspec_mag.sort('V')
calspec_mag = calspec_mag[:-3]

calspec_mag['V'] = calspec_mag['V'].astype(float)
calspec_mag.sort('V')

calspec_mag.show_in_notebook()
